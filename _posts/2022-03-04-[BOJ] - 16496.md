```c++
#include <algorithm>
#include <iostream>
#include <vector>
#include <map>
#include <set>
#include <queue>
#include <cstring>
// #include <bits/stdc++.h>
using namespace std;
#define for0(i, n) for(int i = 0; i < n; i++)
#define for1(i, n) for(int i = 1; i < n; i++)
#define fori(s, e) for(int i = s; i < e; i++)
#define ll long long
#define endl "\n"
#define nulls '\0'
#define dkdk " "
const int MAX = 5e5 + 1;
int n, m, time;

bool cmp(string s1, string s2) {
  return s1+s2 > s2+s1;
}

int main() {
  ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
  cin >> n;
  vector<string> svec;
  svec.resize(n);
  for0(i, n) cin >> svec[i];
  sort(svec.begin(), svec.end(), cmp);
  if(svec[0] == "0") cout << 0;
  else for(string s: svec) cout << s;
  return 0;
}
```
