      for0(i, 3) tmp[i] = cube[3][2-i][h];

  if(face == 'F' || face == 'B') {

    rotate_face(face == 'F' ? 1 : 4, clock);

      for0(i, 3) cube[5][i][h] = store[i], store[i] = tmp[i];
      for0(i, 3) tmp[i] = cube[3][h][2-i];

    int h = face == 'L' ? 2 : 0;

}
