```c++
#include <algorithm>
#include <iostream>
#include <vector>
#include <map>
#include <set>
#include <queue>
#include <cstring>
// #include <bits/stdc++.h>
using namespace std;
#define for0(i, n) for(int i = 0; i < n; i++)
#define for1(i, n) for(int i = 1; i < n; i++)
#define fori(s, e) for(int i = s; i < e; i++)
#define ll long long
#define endl "\n"
#define nulls '\0'
#define dkdk " "
const int MAX = 5e5 + 1;
const int INF = 987654321;
int n;
vector<int> liq;

int main() {
  ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
  cin >> n;
  for0(i, n) { int in; cin >> in; liq.push_back(in); }
  int l = 0, r = n-1, min = INF;
  while(l < r) {
    int v = liq[l] + liq[r];
    if(abs(v) < abs(min)) min = v;
    if(v == 0) break;
    if(v > 0) r--;
    else l++;
  }
  cout << min;
  return 0;
}
```
