vector<vector<int>> child;

int dfs(int idx) {
  if(child[idx].size() == 0) return 0;
  vector<int> vec;
  for(int child_idx: child[idx]) {
    vec.push_back(dfs(child_idx));
  }
  int local_min = 0, t_local = 0;
  sort(vec.begin(), vec.end(), greater<int>());
  for0(i, vec.size()) {
    local_min = max(local_min, i+1 + vec[i]);
  }
  return local_min;
}
