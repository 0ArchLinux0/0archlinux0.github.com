class Solution {
    public boolean isPalindrome(int x) {
        if(x < 0) return false;
        String s = String.valueOf(x);
        int l = s.length();
        boolean isPalindrome = true;
        if(l % 2 == 0) { //0 1 2 3
            for(int i = 0; i <= l/2 - 1; i++) {
                if(s.charAt(i) != s.charAt(l - i - 1)) return false;
            }
        } else {
            for(int i = 0; i <= l/2 - 1; i++) { // 0 1 2 (3)
                if(s.charAt(i) != s.charAt(l - i - 1)) return false;
            }
        }
        return true;
    }
}
