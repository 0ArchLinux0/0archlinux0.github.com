class Solution {
    public int reverse(int x) {
        if(x == 0) return 0;
        int sign = 1;
        long ln = new Long(x), ans = 0;
        if(ln < 0) sign = -1;
        while(ln != 0) {
            ans *= 10;
            ans += ln%10;
            ln = ln/10;
        }
        if(ans >= (long)Integer.MIN_VALUE && ans <= (long)Integer.MAX_VALUE) return (int)ans;
        else return 0;
    }
}
