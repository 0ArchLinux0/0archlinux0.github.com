class Solution {
    public int maxArea(int[] height) {
        int start = 0, end = height.length - 1, max = 0;
        while(start < end) {
            max = Math.max(max, (end - start) * Math.min(height[end], height[start]));
            if(height[start] < height[end]) { // 1 3
                start++;
            } else {
                end--;
            }
        }
        return max;
    }
}
